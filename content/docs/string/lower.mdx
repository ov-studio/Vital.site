---
title: util.string.lower
description: Converts all uppercase letters to lowercase
badge: Shared
---

### Syntax

```lua
local result = util.string.lower(
    input
)
```

### Parameters

| Type | Name | Description |
|------|------|-------------|
| `string` | input | String to convert |

### Returns

| Type | Name | Description |
|------|------|-------------|
| `string` | result | Altered string with all uppercase letters converted to lowercase |

---

### Examples

```lua lineNumbers title="Convert an uppercase string to lowercase"
local result = util.string.lower("HELLO")

core.engine.print("info", result) -- 'hello'
```

```lua lineNumbers title="Convert a mixed-case string to lowercase"
local result = util.string.lower("Hello World")

core.engine.print("info", result) -- 'hello world'
```

```lua lineNumbers title="Already-lowercase strings are unchanged"
local result = util.string.lower("already lowercase")

core.engine.print("info", result) -- 'already lowercase'
```

```lua lineNumbers title="Numbers and special characters are unchanged"
local result = util.string.lower("ABC123!@#")

core.engine.print("info", result) -- 'abc123!@#'
```

```lua lineNumbers title="Perform a case-insensitive comparison"
local str1 = "Hello"
local str2 = "HELLO"

if util.string.lower(str1) == util.string.lower(str2) then
    core.engine.print("info", "Strings are equal (case-insensitive)")
end
```

```lua lineNumbers title="Normalize user input for comparison"
local input = "YES"

if util.string.lower(input) == "yes" then
    core.engine.print("info", "User confirmed")
end
```

```lua lineNumbers title="Convert table keys to lowercase"
local data = {
    Name = "John",
    Age = 25,
    City = "NYC"
}

for key, value in pairs(data) do
    core.engine.print("info", util.string.lower(key), value)
end

--[[
Output: 
'name John'
'age 25'
'city NYC'
]]
```

```lua lineNumbers title="Normalize an email address to lowercase"
local email = "User@Example.COM"
local normalized = util.string.lower(email)

core.engine.print("info", normalized) -- 'user@example.com'
```

```lua lineNumbers title="Strings with only numbers are unchanged"
local result = util.string.lower("123456")

core.engine.print("info", result) -- '123456'
```

```lua lineNumbers title="Lowercase an empty string"
local result = util.string.lower("")

core.engine.print("info", result) -- ''
```

```lua lineNumbers title="Chain lowercase with other string functions"
local result = util.string.lower("HELLO WORLD"):gsub(" ", "_")

core.engine.print("info", result) -- 'hello_world'
```
