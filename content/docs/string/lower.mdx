---
title: string.lower
description: Converts all uppercase letters to lowercase
badge: Shared
---

### Syntax

```lua
local result = string.lower(
    input
)
```

### Parameters

| Type | Name | Description |
|------|------|-------------|
| `string` | input | String to be converted |

### Returns

| Type | Name | Description |
|------|------|-------------|
| `string` | result | Altered string with all uppercase letters converted to lowercase |

<hr/>

### Examples

```lua lineNumbers title="Convert an uppercase string to lowercase"
local result = string.lower("HELLO")

engine.print("info", result) --'hello'
```

```lua lineNumbers title="Convert a mixed-case string to lowercase"
local result = string.lower("Hello World")

engine.print("info", result) --'hello world'
```

```lua lineNumbers title="Already-lowercase strings are unchanged"
local result = string.lower("already lowercase")

engine.print("info", result) --'already lowercase'
```

```lua lineNumbers title="Numbers and special characters are unchanged"
local result = string.lower("ABC123!@#")

engine.print("info", result) --'abc123!@#'
```

```lua lineNumbers title="Perform a case-insensitive comparison"
local str1 = "Hello"
local str2 = "HELLO"

if string.lower(str1) == string.lower(str2) then
    engine.print("info", "Strings are equal (case-insensitive)")
end
```

```lua lineNumbers title="Normalize user input for comparison"
local input = "YES"

if string.lower(input) == "yes" then
    engine.print("info", "User confirmed")
end
```

```lua lineNumbers title="Convert table keys to lowercase"
local data = {
    Name = "John",
    Age = 25,
    City = "NYC"
}

for key, value in pairs(data) do
    engine.print("info", string.lower(key), value)
end

--[[
Output: 
'name John'
'age 25'
'city NYC'
]]
```

```lua lineNumbers title="Normalize an email address to lowercase"
local email = "User@Example.COM"
local normalized = string.lower(email)

engine.print("info", normalized) --'user@example.com'
```

```lua lineNumbers title="Strings with only numbers are unchanged"
local result = string.lower("123456")

engine.print("info", result) --'123456'
```

```lua lineNumbers title="Lowercase an empty string"
local result = string.lower("")

engine.print("info", result) --''
```

```lua lineNumbers title="Chain lowercase with other string functions"
local result = string.lower("HELLO WORLD"):gsub(" ", "_")

engine.print("info", result) --'hello_world'
```
