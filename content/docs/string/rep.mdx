---
title: util.string.rep
description: Repeats a string n times with optional separator
badge: Shared
---

### Syntax

```lua
local result = util.string.rep(
    input,
    count,
    separator = ""
)
```

### Parameters

| Type | Name | Description |
|------|------|-------------|
| `string` | input | String to repeat |
| `int` | count | Number of times to repeat the string |
| `string` | separator | Separator to insert between repetitions |

### Returns

| Type | Name | Description |
|------|------|-------------|
| `string` | result | String formed by repeating the input string specified number of times |

---

### Examples

```lua lineNumbers title="Repeat a string multiple times"
local result = util.string.rep("ab", 3)

core.engine.print("info", result) -- 'ababab'
```

```lua lineNumbers title="Repeat a string with a separator"
local result = util.string.rep("hi", 3, "-")

core.engine.print("info", result) -- 'hi-hi-hi'
```

```lua lineNumbers title="Create a padding string"
local padding = util.string.rep(" ", 10)

core.engine.print("info", "["..padding.."]") -- '[          ]'
```

```lua lineNumbers title="Return empty string with zero repetitions"
local result = util.string.rep("test", 0)

core.engine.print("info", result) -- ''
```

```lua lineNumbers title="Return the original string with one repetition"
local result = util.string.rep("test", 1)

core.engine.print("info", result) -- 'test'
```

```lua lineNumbers title="Repeat using method syntax"
local result = util.string.rep("*", 5)

core.engine.print("info", result) -- '*****'
```

```lua lineNumbers title="Create a separator line"
local separator = util.string.rep("=", 50)

core.engine.print("info", separator)
```

```lua lineNumbers title="Create a visual progress bar"
local bar = util.string.rep("█", 10)

core.engine.print("info", bar) -- '██████████'
```

```lua lineNumbers title="Repeat words with a space separator"
local result = util.string.rep("word", 4, " ")

core.engine.print("info", result) -- 'word word word word'
```

```lua lineNumbers title="Create indentation with repeated spaces"
local indent = util.string.rep("  ", 3)

core.engine.print("info", indent.."code") -- '      code'
```

```lua lineNumbers title="Build an ASCII art pattern"
local pattern = util.string.rep("* ", 5)

core.engine.print("info", pattern) -- '* * * * * '
```

```lua lineNumbers title="Create a repeated table row"
local cells = util.string.rep("| Cell ", 4)

core.engine.print("info", cells.."|") -- '| Cell | Cell | Cell | Cell |'
```

```lua lineNumbers title="Generate a fixed-length test string"
local data = util.string.rep("A", 100)

core.engine.print("info", #data) -- 100
```

```lua lineNumbers title="Repeat a multi-character pattern"
local pattern = util.string.rep("abc", 3)

core.engine.print("info", pattern) -- 'abcabcabc'
```

```lua lineNumbers title="Repeat fields with a comma separator"
local csv = util.string.rep("field", 5, ",")

core.engine.print("info", csv) -- 'field,field,field,field,field'
```
