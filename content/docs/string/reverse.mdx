---
title: string.reverse
description: Returns a string with characters in reverse order
badge: Shared
---

### Syntax

```lua
local result = string.reverse(input)
```

### Parameters

| Type | Name | Description |
|------|------|-------------|
| `string` | input | String to be reversed |

### Returns

| Type | Name | Description |
|------|------|-------------|
| `string` | result | String with all characters reversed |

<hr/>

### Examples

```lua lineNumbers title="Reverse a word"
local result = string.reverse("hello")

engine.print("info", result) --'olleh'
```

```lua lineNumbers title="Reverse an entire sentence"
local result = string.reverse("hello world")

engine.print("info", result) --'dlrow olleh'
```

```lua lineNumbers title="Reverse a numeric string"
local result = string.reverse("12345")

engine.print("info", result) --'54321'
```

```lua lineNumbers title="Reversing a single character returns itself"
local result = string.reverse("a")

engine.print("info", result) --'a'
```

```lua lineNumbers title="Reversing an empty string returns empty"
local result = string.reverse("")

engine.print("info", result) --''
```

```lua lineNumbers title="Check if a string is a palindrome"
local function isPalindrome(str)
    return str == string.reverse(str)
end

engine.print("info", isPalindrome("racecar")) --true
engine.print("info", isPalindrome("hello")) --false
```

```lua lineNumbers title="Reverse a string and convert to uppercase"
local result = string.upper(string.reverse("hello"))

engine.print("info", result) --'OLLEH'
```

```lua lineNumbers title="Reverse a string with special characters"
local result = string.reverse("hello!")

engine.print("info", result) --'!olleh'
```

```lua lineNumbers title="Double-reversing returns the original string"
local original = "test"
local reversed = string.reverse(string.reverse(original))

engine.print("info", reversed == original) --true
```

```lua lineNumbers title="Reverse each word in a sentence individually"
local sentence = "hello world"
local reversed = string.gsub(sentence, "%S+", function(word)
    return string.reverse(word)
end)

engine.print("info", reversed) --'olleh dlrow'
```

```lua lineNumbers title="Create a mirror effect with a string"
local text = "ABC"
local mirror = text.."|"..string.reverse(text)

engine.print("info", mirror) --'ABC|CBA'
```

```lua lineNumbers title="Reverse a binary string"
local binary = "1010"
local reversed = string.reverse(binary)

engine.print("info", reversed) --'0101'
```

```lua lineNumbers title="Reverse a number represented as a string"
local num = '123456'

engine.print("info", string.reverse(num)) --'654321'
```

```lua lineNumbers title="Iterate over a string from end to start"
local data = 'ABCDEF'

for i = 1, #data do
    local char = string.sub(string.reverse(data), i, i)
    engine.print("info", char)
end

--[[
Output: 
'F'
'E'
'D'
'C'
'B'
'A'
]]
```
