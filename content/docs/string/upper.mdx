---
title: string.upper
description: Converts all lowercase letters to uppercase
badge: Shared
---

### Syntax

```lua
local result = string.upper(input)
```

### Parameters

| Type | Name | Description |
|------|------|-------------|
| `string` | input | String to be converted |

### Returns

| Type | Name | Description |
|------|------|-------------|
| `string` | result | Altered string with all uppercase letters converted to uppercase |

<hr/>

### Examples

```lua lineNumbers title="Convert a lowercase string to uppercase"
--Convert to uppercase
local result = string.upper("hello")
engine.print("info", result) --HELLO
```

```lua lineNumbers title="Convert a mixed-case string to uppercase"
--Mixed case
local result = string.upper("Hello World")
engine.print("info", result) --HELLO WORLD
```

```lua lineNumbers title="Already-uppercase strings are unchanged"
--Already uppercase (no change)
local result = string.upper("ALREADY UPPERCASE")
engine.print("info", result) --ALREADY UPPERCASE
```

```lua lineNumbers title="Numbers and special characters are unchanged"
--Numbers and special characters unchanged
local result = string.upper("abc123!@#")
engine.print("info", result) --ABC123!@#
```

```lua lineNumbers title="Perform a case-insensitive comparison"
--Case-insensitive comparison
local str1 = "Hello"
local str2 = "hello"
if string.upper(str1) == string.upper(str2) then
    engine.print("info", "Strings are equal (case-insensitive)")
end
```

```lua lineNumbers title="Normalize user input for comparison"
--Normalize user input
local input = "yes"
if string.upper(input) == "YES" then
    engine.print("info", "User confirmed")
end
```

```lua lineNumbers title="Create a constant name from a variable name"
--Create constant names
local varName = "max_value"
local constant = string.upper(varName)
engine.print("info", constant) --MAX_VALUE
```

```lua lineNumbers title="Emphasize text with uppercase"
--Emphasize text
local text = "warning"
local emphasized = string.upper(text)
engine.print("info", emphasized) --WARNING
```

```lua lineNumbers title="Convert a status string to an enum value"
--Convert enum values
local status = "active"
local enum = string.upper(status)
engine.print("info", enum) --ACTIVE
```

```lua lineNumbers title="Strings with only numbers are unchanged"
--String with only numbers (no change)
local result = string.upper("123456")
engine.print("info", result) --123456
```

```lua lineNumbers title="Uppercase an empty string"
--Empty string
local result = string.upper("")
engine.print("info", result) --
```

```lua lineNumbers title="Convert a title to an uppercase header"
--Title to header
local title = "my document title"
local header = string.upper(title)
engine.print("info", header) --MY DOCUMENT TITLE
```

```lua lineNumbers title="Convert a column name to SQL-style uppercase"
--SQL-like syntax
local column = "user_id"
engine.print("info", string.upper(column)) --USER_ID
```

```lua lineNumbers title="Compare two strings ignoring case"
--Compare ignore case
local function equalsIgnoreCase(a, b)
    return string.upper(a) == string.upper(b)
end
engine.print("info", equalsIgnoreCase("Test", "test")) --true
```

```lua lineNumbers title="Format a command string in uppercase"
--Format command
local cmd = "help"
engine.print("info", "Command: "..string.upper(cmd)) --Command: HELP
```

```lua lineNumbers title="Build an acronym from a phrase"
--Acronym from words
local text = "hypertext markup language"
local acronym = ""
for word in string.gmatch(text, "%S+") do
    acronym = acronym..string.upper(string.sub(word, 1, 1))
end
engine.print("info", acronym) --HML
```

```lua lineNumbers title="Convert to shouty snake case"
--Shouty snake case
local name = "my_variable_name"
local shouty = string.upper(name)
engine.print("info", shouty) --MY_VARIABLE_NAME
```
