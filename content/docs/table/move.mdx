---
title: table.move
description: Moves elements from one table to another
badge: Shared
---

### Syntax

```lua
local result = table.move(from, start_at, end_at, move_at, to = from)
```

### Parameters

| Type | Name | Description |
|------|------|-------------|
| `table` | from | Source table to copy from |
| `int` | start_at | Starting index in from table |
| `int` | end_at | Ending index in from table |
| `int` | move_at | Starting index in destination table |
| `table` | to | Destination table (defaults to from table) |

### Returns

| Type | Name | Description |
|------|------|-------------|
| `table` | result | Modified destination table |

<hr/>

### Example

```lua
--Copy elements to another table
local src = {1, 2, 3, 4, 5}
local dest = {}
table.move(src, 1, 3, 1, dest)
engine.print(table.concat(dest, ", ")) --1, 2, 3
```

```lua
--Move elements within same table
local t = {10, 20, 30, 40, 50}
table.move(t, 1, 2, 4)
engine.print(table.concat(t, ", ")) --10, 20, 30, 10, 20
```

```lua
--Shift elements right
local t = {1, 2, 3, 4}
table.move(t, 1, 4, 2)
engine.print(table.concat(t, ", ")) --1, 1, 2, 3, 4
```

```lua
--Copy subset to new position
local t = {"a", "b", "c", "d", "e"}
table.move(t, 2, 4, 6)
engine.print(table.concat(t, ", ")) --a, b, c, d, e, b, c, d
```

```lua
--Clone entire table
local original = {1, 2, 3}
local clone = {}
table.move(original, 1, #original, 1, clone)
engine.print(table.concat(clone, ", ")) --1, 2, 3
```

```lua
--Insert at beginning (shift right)
local t = {2, 3, 4}
table.move(t, 1, #t, 2)
t[1] = 1
engine.print(table.concat(t, ", ")) --1, 2, 3, 4
```

```lua
--Append from another table
local t1 = {1, 2, 3}
local t2 = {4, 5, 6}
table.move(t2, 1, #t2, #t1 + 1, t1)
engine.print(table.concat(t1, ", ")) --1, 2, 3, 4, 5, 6
```

```lua
--Remove by shifting left
local t = {1, 2, 3, 4, 5}
table.move(t, 3, 5, 2)
table.remove(t)
table.remove(t)
engine.print(table.concat(t, ", ")) --1, 3, 4, 5
```

```lua
--Duplicate range
local t = {"x", "y", "z"}
table.move(t, 1, 3, 4)
engine.print(table.concat(t, ", ")) --x, y, z, x, y, z
```

```lua
--Copy partial range
local src = {10, 20, 30, 40, 50}
local dest = {0, 0, 0, 0, 0}
table.move(src, 2, 4, 1, dest)
engine.print(table.concat(dest, ", ")) --20, 30, 40, 0, 0
```

```lua
--Reverse copy
local src = {1, 2, 3, 4}
local dest = {}
for i = #src, 1, -1 do
    table.move(src, i, i, #src - i + 1, dest)
end
engine.print(table.concat(dest, ", ")) --4, 3, 2, 1
```

```lua
--Efficient array extension
local base = {1, 2, 3}
local extension = {4, 5, 6, 7, 8}
table.move(extension, 1, #extension, #base + 1, base)
engine.print(table.concat(base, ", ")) --1, 2, 3, 4, 5, 6, 7, 8
```
