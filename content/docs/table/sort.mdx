---
title: table.sort
description: Sorts table elements in-place
badge: Shared
---

### Syntax

```lua
table.sort(input, comparison = nil)
```

### Parameters

| Type | Name | Description |
|------|------|-------------|
| `table` | input | Table to sort |
| `function` | comparison | Comparison function `compare(a, b)` - used for manipulating sorting algorithm |

### Returns

| Type | Name | Description |
|------|------|-------------|

<hr/>

### Example

```lua
--Sort numbers ascending (default)
local nums = {5, 2, 8, 1, 9}
table.sort(nums)
engine.print(table.concat(nums, ", ")) --1, 2, 5, 8, 9
```

```lua
--Sort numbers descending
local nums = {5, 2, 8, 1, 9}
table.sort(nums, function(a, b) return a > b end)
engine.print(table.concat(nums, ", ")) --9, 8, 5, 2, 1
```

```lua
--Sort strings alphabetically
local words = {"zebra", "apple", "mango", "banana"}
table.sort(words)
engine.print(table.concat(words, ", ")) --apple, banana, mango, zebra
```

```lua
--Sort strings by length
local words = {"hi", "hello", "a", "world"}
table.sort(words, function(a, b) return #a < #b end)
engine.print(table.concat(words, ", ")) --a, hi, hello, world
```

```lua
--Sort case-insensitive
local names = {"Bob", "alice", "Charlie", "david"}
table.sort(names, function(a, b) 
    return string.lower(a) < string.lower(b) 
end)
engine.print(table.concat(names, ", ")) --alice, Bob, Charlie, david
```

```lua
--Sort table of tables by property
local players = {
    {name = "Alice", score = 100},
    {name = "Bob", score = 150},
    {name = "Charlie", score = 75}
}
table.sort(players, function(a, b) return a.score > b.score end)
for _, player in ipairs(players) do
    engine.print(player.name, player.score)
end
--Output: Bob 150, Alice 100, Charlie 75
```

```lua
--Sort mixed case strings
local items = {"Apple", "banana", "Cherry", "date"}
table.sort(items, function(a, b)
    return string.lower(a) < string.lower(b)
end)
engine.print(table.concat(items, ", "))
```

```lua
--Custom sort: even numbers first
local nums = {1, 2, 3, 4, 5, 6}
table.sort(nums, function(a, b)
    if (a%2 == 0) == (b%2 == 0) then
        return a < b
    end
    return (a%2 == 0)
end)
engine.print(table.concat(nums, ", ")) --2, 4, 6, 1, 3, 5
```
